package com.exe1;

import java.util.Scanner;

class UserDate{
	static boolean b=false;
	
	static boolean validateDate(String date) {
		String dateFormat = "\\d{1,2}-\\d{1,2}-\\d{4}";
		
		return b=date.matches(dateFormat);
		
	}
	void output() {
		if(b==true)
			System.out.println("Valid");
		else
			System.out.println("Invalid");
	}
}

public class Example1 {
	
	public static void main(String args[]){
		UserDate u=new UserDate();
		Scanner s=new Scanner(System.in);
		String date=s.nextLine();
		u.validateDate(date);
		u.output();
	}
}

package com.exe1;

import java.util.Scanner;

class Shape{
	protected String shapeName;

	public Shape(String shapeName) {
		super();
		this.shapeName = shapeName;
	}	
	double calculateArea() {
		return 0;
	}
}
class Square extends Shape{	
	int side;	
	public Square(String shapeName,int side) {
		super(shapeName);
		this.shapeName=shapeName;
		this.side=side;
	}
	double calculateArea() {
		 return side*side;
	}
	void output() {
		System.out.println("Area of "+shapeName+" is:"+calculateArea());
	}		
}
class Rectangle extends Shape{
	private int length;
	private int breadth;
	public Rectangle(String shapeName, int length, int breadth) {
		super(shapeName);
		this.shapeName=shapeName;
		this.length = length;
		this.breadth = breadth;
	}
	
	double calculateArea() {
		return length*breadth;
	}
	void output() {
		System.out.println("Area of "+shapeName+" is:"+calculateArea());
	}
}
class Circle extends Shape{
	private int radius;
	public Circle(String shapeName, int radius) {
		super(shapeName);
		this.shapeName=shapeName;
		this.radius = radius;
	}
	double calculateArea() {
		return 3.14*radius*radius;
	}
	void output() {
		System.out.println("Area of "+shapeName+" is:"+calculateArea());
	}
	
}
